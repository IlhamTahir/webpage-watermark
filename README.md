Webpage Watermark
===

## 调用方法

```
let Watermark = require('webpage-watermark');

new Watermark({
    fontFamily: 'Arial',
    fontStyle: 'normal',
    fontVariant: 'normal',
    fontWeight: 'normal',
    fontSize: '14px',
    color: '#000',
    id: 'watermark',
    text: 'Watermark',
    density: 100,
    clarity: 2,
    zIndex: -1,
    supportTip: '您的浏览器不支持Canvas'
})
```
